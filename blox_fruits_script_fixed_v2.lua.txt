-- Animation System
local function animateButton()
    local tween1 = game:GetService("TweenService"):Create(
        UIScale,
        TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut),
        {Scale = 0.8}
    )
    tween1:Play()
    tween1.Completed:Connect(function()
        local tween2 = game:GetService("TweenService"):Create(
            UIScale,
            TweenInfo.new(0.1, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
            {Scale = 1.1}
        )
        tween2:Play()
        tween2.Completed:Connect(function()
            local tween3 = game:GetService("TweenService"):Create(
                UIScale,
                TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
                {Scale = 1}
            )
            tween3:Play()
        end)
    end)
end

-- Advanced Teleportation System
local function TP(P1)
    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end
    
    local Distance = (P1.Position - character.HumanoidRootPart.Position).Magnitude
    local Speed
    
    if Distance < 250 then
        Speed = 5000
    elseif Distance < 500 then
        Speed = 650
    elseif Distance < 1000 then
        Speed = 350
    elseif Distance >= 1000 then
        Speed = 250
    end
    
    local tween = TweenService:Create(
        character.HumanoidRootPart,
        TweenInfo.new(Distance/Speed, Enum.EasingStyle.Linear),
        {CFrame = P1}
    )
    tween:Play()
    return tween
end

local function TP2(P1)
    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end
    
    local Distance = (P1.Position - character.HumanoidRootPart.Position).Magnitude
    local Speed
    
    if Distance < 150 then
        Speed = 5000
    elseif Distance < 200 then
        Speed = 1500
    elseif Distance < 300 then
        Speed = 800
    elseif Distance < 500 then
        Speed = 500
    elseif Distance < 1000 then
        Speed = 250
    elseif Distance >= 1000 then
        Speed = 250
    end
    
    _G.Clip = true
    local tween = TweenService:Create(
        character.HumanoidRootPart,
        TweenInfo.new(Distance/Speed, Enum.EasingStyle.Linear),
        {CFrame = P1}
    )
    tween:Play()
    wait(Distance/Speed)
    _G.Clip = false
    return tween
end

-- Advanced Fast Attack System
local function AttackNoCD()
    if not getgenv().Settings.FastAttack then return end
    
    local combat = require(ReplicatedStorage.CombatFramework.CombatFramework)
    local activeController = combat.activeController
    if activeController and activeController.equipped then
        for i = 1, 1 do
            local bladehit = require(ReplicatedStorage.CombatFramework.RigLib).getBladeHits(
                player.Character,
                {player.Character.HumanoidRootPart},
                60
            )
            local cframe = require(ReplicatedStorage.CombatFramework.RigLib).raycast(
                player.Character.HumanoidRootPart.CFrame,
                Vector3.new(0, -50, 0),
                {player.Character}
            )
            ReplicatedStorage.RigControllerEvent:FireServer("weaponChange", tostring(activeController.blades[1].Parent))
            ReplicatedStorage.RigControllerEvent:FireServer("hit", bladehit, i, "")
        end
    end
end

-- Weapon System
local function getPlayerWeapons()
    local weapons = {}
    for _, tool in pairs(player.Backpack:GetChildren()) do
        if tool:IsA("Tool") then
            table.insert(weapons, tool.Name)
        end
    end
    for _, tool in pairs(player.Character:GetChildren()) do
        if tool:IsA("Tool") then
            table.insert(weapons, tool.Name)
        end
    end
    return weapons
end

local function EquipWeapon(weaponName)
    if not weaponName or weaponName == "" then return end
    
    if player.Backpack:FindFirstChild(weaponName) then
        local weapon = player.Backpack:FindFirstChild(weaponName)
        player.Character.Humanoid:EquipTool(weapon)
    elseif player.Character:FindFirstChild(weaponName) then
        -- Already equipped
    end
end

-- Fruit ESP System with Distance (Red Color)
local function createFruitESP()
    if not getgenv().Settings.FruitESP then return end
    
    for _, fruit in pairs(game.Workspace:GetChildren()) do
        if string.find(fruit.Name, "Fruit") and fruit:FindFirstChild("Handle") then
            if not fruit.Handle:FindFirstChild("FruitESP") then
                local billboardGui = Instance.new("BillboardGui")
                local textLabel = Instance.new("TextLabel")
                
                billboardGui.Name = "FruitESP"
                billboardGui.Parent = fruit.Handle
                billboardGui.Size = UDim2.new(0, 200, 0, 50)
                billboardGui.StudsOffset = Vector3.new(0, 2, 0)
                billboardGui.AlwaysOnTop = true
                
                textLabel.Parent = billboardGui
                textLabel.Size = UDim2.new(1, 0, 1, 0)
                textLabel.BackgroundTransparency = 1
                textLabel.Text = fruit.Name .. "\n[" .. math.floor((fruit.Handle.Position - player.Character.HumanoidRootPart.Position).Magnitude) .. " studs]"
                textLabel.TextColor3 = Color3.fromRGB(255, 0, 0) -- Red color
                textLabel.TextStrokeTransparency = 0
                textLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
                textLabel.Font = Enum.Font.GothamBold
                textLabel.TextSize = 14
                
                spawn(function()
                    while fruit and fruit.Parent and textLabel do
                        wait(0.5)
                        if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                            textLabel.Text = fruit.Name .. "\n[" .. math.floor((fruit.Handle.Position - player.Character.HumanoidRootPart.Position).Magnitude) .. " studs]"
                        end
                    end
                end)
            end
        end
    end
end
