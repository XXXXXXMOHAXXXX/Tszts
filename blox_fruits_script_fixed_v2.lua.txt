-- Professional Teleport Function
function Teleport(pos)
    if not Player.Character or not Player.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    
    local Distance = (pos.Position - Player.Character.HumanoidRootPart.Position).Magnitude
    local Speed = 300
    
    if Distance < 10 then
        Player.Character.HumanoidRootPart.CFrame = pos
        return
    end
    
    local Time = Distance / Speed
    local TweenInfo = TweenInfo.new(Time, Enum.EasingStyle.Linear)
    local Tween = TweenService:Create(Player.Character.HumanoidRootPart, TweenInfo, {CFrame = pos})
    
    Tween:Play()
    return Tween
end

-- Quest System Functions (WORKING LOGIC)
function CheckQuest()
    CheckLevel()
    if Player.PlayerGui.Main.Quest.Visible == false then
        return false
    end
    local QuestTitle = Player.PlayerGui.Main.Quest.Container.QuestTitle.Title.Text
    if not string.find(QuestTitle, NameMon) then
        return false
    end
    return true
end

function GetQuest()
    CheckLevel()
    Teleport(CFrameQuest)
    wait(1)
    
    pcall(function()
        local questNPC = Workspace.NPCs:FindFirstChild(NameQuest)
        if questNPC and questNPC:FindFirstChild("HumanoidRootPart") then
            if questNPC.HumanoidRootPart:FindFirstChild("ProximityPrompt") then
                fireproximityprompt(questNPC.HumanoidRootPart.ProximityPrompt)
                wait(0.5)
            end
            
            local remote = ReplicatedStorage.Remotes.CommF_
            remote:InvokeServer("StartQuest", NameQuest, LevelQuest)
            wait(1)
        end
    end)
end

-- Weapon System (PROPER LOGIC)
function EquipWeapon()
    if getgenv().SelectedWeapon == "" then
        for i, v in pairs(Player.Backpack:GetChildren()) do
            if v:IsA("Tool") then
                Player.Character.Humanoid:EquipTool(v)
                getgenv().SelectedWeapon = v.Name
                break
            end
        end
    else
        local weapon = Player.Backpack:FindFirstChild(getgenv().SelectedWeapon)
        if weapon then
            Player.Character.Humanoid:EquipTool(weapon)
        end
    end
end

-- Fast Attack System (WORKING REMOTE EVENTS)
function FastAttack()
    if not getgenv().FastAttack then return end
    
    pcall(function()
        -- Combat Framework Attack
        local combat = ReplicatedStorage:FindFirstChild("CombatFramework")
        if combat then
            local rigController = combat:FindFirstChild("RigController")
            if rigController then
                rigController.timeStamp = 0
                rigController:attack()
            end
        end
        
        -- Direct Remote Events for M1 Attacks
        for i, v in pairs(Player.Character:GetChildren()) do
            if v:IsA("Tool") then
                -- Main attack remotes
                ReplicatedStorage.RigControllerEvent:FireServer("weaponChange", v.Name)
                ReplicatedStorage.RigControllerEvent:FireServer("hit", {}, {}, "")
                
                -- Additional attack remotes for better damage
                local args = {
                    [1] = 1,
                    [2] = Player.Character.HumanoidRootPart.CFrame,
                    [3] = "rbxassetid://131961136",
                    [4] = {
                        ["Position"] = Player.Character.HumanoidRootPart.Position,
                        ["Normal"] = Vector3.new(0, 1, 0),
                        ["Part"] = workspace.Enemies:FindFirstChild(NameMon)
                    }
                }
                
                ReplicatedStorage.Remotes.CommF_:InvokeServer("TAP", args)
                break
            end
        end
    end)
end

-- Combat System (COMPLETE WORKING LOGIC)
function AttackMobs()
    for i, v in pairs(Workspace.Enemies:GetChildren()) do
        if v.Name == NameMon and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
            if v:FindFirstChild("HumanoidRootPart") then
                -- Teleport ABOVE the mob for better positioning
                Teleport(v.HumanoidRootPart.CFrame * CFrame.new(0, 50, 0))
                
                -- Equip weapon first
                EquipWeapon()
                
                -- Enable auto haki
                if getgenv().AutoHaki then
                    ReplicatedStorage.Remotes.CommF_:InvokeServer("Buso")
                end
                
                -- Fast attack the mob
                FastAttack()
                
                -- Large Hitbox for easier targeting
                if v:FindFirstChild("HumanoidRootPart") then
                    v.HumanoidRootPart.Size = Vector3.new(60, 60, 60)
                    v.HumanoidRootPart.Transparency = 0.8
                    v.HumanoidRootPart.CanCollide = false
                end
                
                -- Bring mobs system (WORKING MAGNETIZATION)
                if getgenv().BringMobs then
                    for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
                        if enemy.Name == NameMon and enemy ~= v and enemy:FindFirstChild("HumanoidRootPart") and enemy.Humanoid.Health > 0 then
                            enemy.HumanoidRootPart.CFrame = v.HumanoidRootPart.CFrame
                            enemy.HumanoidRootPart.CanCollide = false
                            enemy.HumanoidRootPart.Size = Vector3.new(60, 60, 60)
                            if enemy:FindFirstChild("Humanoid") then
                                enemy.Humanoid.WalkSpeed = 0
                                enemy.Humanoid.JumpPower = 0
                            end
                        end
                    end
                end
                
                break
            end
        end
    end
end
