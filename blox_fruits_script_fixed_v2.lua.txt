-- Weapon Functions
local function getPlayerWeapons()
    local weapons = {}
    for _, tool in pairs(LocalPlayer.Backpack:GetChildren()) do
        if tool:IsA("Tool") then
            table.insert(weapons, tool.Name)
        end
    end
    if LocalPlayer.Character then
        for _, tool in pairs(LocalPlayer.Character:GetChildren()) do
            if tool:IsA("Tool") then
                table.insert(weapons, tool.Name)
            end
        end
    end
    return weapons
end

local function equipWeapon(weaponName)
    if LocalPlayer.Backpack:FindFirstChild(weaponName) then
        local tool = LocalPlayer.Backpack:FindFirstChild(weaponName)
        Humanoid:EquipTool(tool)
    end
end

-- Fast Attack System
local fastAttackConnections = {}
local function fastAttack()
    if _G.FastAttack then
        local ac = RunService.Heartbeat:Connect(function()
            pcall(function()
                if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
                    local tool = LocalPlayer.Character:FindFirstChildOfClass("Tool")
                    if tool and tool:FindFirstChild("Handle") then
                        tool:Activate()
                        -- Simulate melee attacks
                        game:GetService("VirtualUser"):CaptureController()
                        game:GetService("VirtualUser"):Button1Down(Vector2.new(0,0))
                        game:GetService("VirtualUser"):Button1Up(Vector2.new(0,0))
                        
                        -- Fire remote events for weapons
                        for _, v in pairs(tool:GetDescendants()) do
                            if v:IsA("RemoteEvent") then
                                v:FireServer(pos)
                            end
                        end
                    end
                end
            end)
        end)
        fastAttackConnections[#fastAttackConnections + 1] = ac
    else
        for i, v in pairs(fastAttackConnections) do
            v:Disconnect()
        end
        fastAttackConnections = {}
    end
end

-- Mob Bringing and Hitbox System
local function bringMobs()
    spawn(function()
        while _G.BringMobs do
            pcall(function()
                for _, mob in pairs(Workspace.Enemies:GetChildren()) do
                    if mob:FindFirstChild("Humanoid") and mob:FindFirstChild("HumanoidRootPart") and mob.Humanoid.Health > 0 then
                        if string.find(mob.Name, QuestData.Monster) and (mob.HumanoidRootPart.Position - HumanoidRootPart.Position).Magnitude <= 400 then
                            -- Bring mob close and expand hitbox
                            mob.HumanoidRootPart.CFrame = HumanoidRootPart.CFrame * CFrame.new(0, -40, 0)
                            mob.HumanoidRootPart.CanCollide = false
                            mob.Humanoid.WalkSpeed = 0
                            mob.Humanoid.JumpPower = 0
                            mob.Humanoid.PlatformStand = true
                            mob.Humanoid.Sit = true
                            
                            -- Massive hitbox expansion
                            mob.HumanoidRootPart.Size = Vector3.new(200, 200, 200)
                            mob.HumanoidRootPart.Transparency = 0.8
                            
                            -- Disable mob movement completely
                            if mob:FindFirstChild("BodyVelocity") then
                                mob.BodyVelocity:Destroy()
                            end
                            if mob:FindFirstChild("BodyPosition") then
                                mob.BodyPosition:Destroy()
                            end
                            if mob:FindFirstChild("BodyAngularVelocity") then
                                mob.BodyAngularVelocity:Destroy()
                            end
                        end
                    end
                end
            end)
            wait(0.1)
        end
    end)
end

-- Movement System
local function tweenToPosition(targetCFrame)
    local distance = (HumanoidRootPart.Position - targetCFrame.Position).Magnitude
    local speed = 300
    local tweenTime = distance / speed
    
    local tween = TweenService:Create(
        HumanoidRootPart,
        TweenInfo.new(tweenTime, Enum.EasingStyle.Linear),
        {CFrame = targetCFrame}
    )
    tween:Play()
    return tween
end

-- Auto Farm Level System
local function autoFarmLevel()
    spawn(function()
        while _G.AutoFarmLevel do
            pcall(function()
                getQuestInfo()
                
                -- Check if quest is active
                if not LocalPlayer.PlayerGui.Main.Quest.Visible then
                    _G.BringMobs = false
                    -- Go to quest giver
                    tweenToPosition(QuestData.QuestCFrame)
                    wait(2)
                    
                    -- Accept quest
                    ReplicatedStorage.Remotes.CommF_:InvokeServer("StartQuest", QuestData.QuestName, QuestData.QuestNumber)
                    wait(1)
                end
                
                -- Enable mob bringing and go to monster location
                _G.BringMobs = true
                tweenToPosition(QuestData.MonsterCFrame)
                wait(2)
                
                -- Check for quest monsters and attack
                local questActive = LocalPlayer.PlayerGui.Main.Quest.Visible
                if questActive then
                    for _, monster in pairs(Workspace.Enemies:GetChildren()) do
                        if string.find(monster.Name, QuestData.Monster) and monster:FindFirstChild("Humanoid") and monster.Humanoid.Health > 0 then
                            repeat
                                wait(0.1)
                                
                                -- Ensure weapon is equipped
                                if _G.SelectedWeapon ~= "" then
                                    equipWeapon(_G.SelectedWeapon)
                                end
                                
                                -- Position above monster for optimal attack
                                if monster and monster:FindFirstChild("HumanoidRootPart") then
                                    HumanoidRootPart.CFrame = monster.HumanoidRootPart.CFrame * CFrame.new(0, 35, 0)
                                    
                                    -- Expand monster hitbox for easier hits
                                    monster.HumanoidRootPart.Size = Vector3.new(200, 200, 200)
                                    monster.HumanoidRootPart.CanCollide = false
                                    monster.HumanoidRootPart.Transparency = 0.8
                                    monster.Humanoid.PlatformStand = true
                                    monster.Humanoid.Sit = true
                                    monster.Humanoid.WalkSpeed = 0
                                    monster.Humanoid.JumpPower = 0
                                    
                                    -- Disable monster body movers
                                    for _, v in pairs(monster:GetDescendants()) do
                                        if v:IsA("BodyVelocity") or v:IsA("BodyPosition") or v:IsA("BodyAngularVelocity") then
                                            v:Destroy()
                                        end
                                    end
                                    
                                    -- Attack with melee weapon
                                    local tool = LocalPlayer.Character:FindFirstChildOfClass("Tool")
                                    if tool then
                                        -- Activate tool for damage
                                        tool:Activate()
                                        
                                        -- Simulate M1 melee attacks
                                        VirtualUser:CaptureController()
                                        VirtualUser:Button1Down(Vector2.new(0, 0))
                                        wait(0.01)
                                        VirtualUser:Button1Up(Vector2.new(0, 0))
                                        
                                        -- Fire weapon remote events
                                        for _, remote in pairs(tool:GetDescendants()) do
                                            if remote:IsA("RemoteEvent") and remote.Name == "RemoteEvent" then
                                                remote:FireServer()
                                            end
                                        end
                                        
                                        -- Additional damage calls for swords/melee
                                        if tool:FindFirstChild("Handle") then
                                            for _, child in pairs(tool.Handle:GetChildren()) do
                                                if child:IsA("RemoteEvent") then
                                                    child:FireServer()
                                                end
                                            end
                                        end
                                    end
                                end
                                
                            until not monster.Parent or monster.Humanoid.Health <= 0 or not _G.AutoFarmLevel or not LocalPlayer.PlayerGui.Main.Quest.Visible
                        end
                    end
                end
            end)
            wait(0.5)
        end
    end)
end

                        billboard.StudsOffset = Vector3.new(0, 3, 0)
                        billboard.AlwaysOnTop = true

                        label.Parent = billboard
                        label.Size = UDim2.new(1, 0, 1, 0)
                        label.BackgroundTransparency = 1
                        label.Text = fruit.Name
                        label.TextColor3 = Color3.new(1, 0, 0)
                        label.TextScaled = true
                        label.Font = Enum.Font.GothamBold
                    end
                end
            end
            wait(1)
        end
    end)
end

local function bringFruit()
    spawn(function()
        while _G.BringFruit do
            for _, fruit in pairs(Workspace:GetChildren()) do
                if string.find(fruit.Name, "Fruit") and fruit:FindFirstChild("Handle") then
                    fruit.Handle.CFrame = HumanoidRootPart.CFrame
                end
            end
            wait(0.5)
        end
    end)
end

local function storeFruit()
    if LocalPlayer.Character:FindFirstChild("Fruit") then
        ReplicatedStorage.Remotes.CommF_:InvokeServer("StoreFruit", LocalPlayer.Character.Fruit.Name)
    end
end
