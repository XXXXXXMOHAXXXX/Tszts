-- Tween Function
function totarget(CFgo)
    local tween_s = game:GetService("TweenService")
    local info = TweenInfo.new((Player.Character.HumanoidRootPart.Position - CFgo.Position).Magnitude / 325, Enum.EasingStyle.Linear)
    local tween = tween_s:Create(Player.Character.HumanoidRootPart, info, {CFrame = CFgo})
    tween:Play()
    
    local function stop()
        tween:Cancel()
        getgenv().StopTween = true
    end
    
    if getgenv().StopTween == true then
        stop()
    end
end

function StopTween()
    getgenv().StopTween = true
end

-- Equip Weapon Function
function EquipWeapon(ToolSe)
    if Player.Character and Player.Character:FindFirstChild("Humanoid") then
        for i,v in pairs(Player.Character:GetChildren()) do
            if v.Name == ToolSe then
                return
            end
        end
        
        for i,v in pairs(Player.Backpack:GetChildren()) do
            if v.Name == ToolSe then
                Player.Character.Humanoid:EquipTool(v)
                return
            end
        end
    end
end

-- Bring Mobs Function  
function BringMobsToPlayer()
    for i,v in pairs(Workspace.Enemies:GetChildren()) do
        if v.Name == Mon and v:FindFirstChild("Humanoid") and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
            if getgenv().BringMobs then
                v.HumanoidRootPart.CFrame = Player.Character.HumanoidRootPart.CFrame * CFrame.new(0, 0, -5)
                v.Humanoid.PlatformStand = true
                v.Humanoid.Sit = true
                v.HumanoidRootPart.CanCollide = false
                v.HumanoidRootPart.Size = Vector3.new(50, 50, 50)
            end
        end
    end
end
